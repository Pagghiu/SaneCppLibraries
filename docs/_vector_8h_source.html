294inline SC::SegmentHeader* SC::VectorAllocator::reallocate(SegmentHeader* oldHeader, size_t newSize)
295{
296 if (newSize > SegmentHeader::MaxValue)
297 {
298 return nullptr;
299 }
300 SegmentHeader* newHeader;
301 if (oldHeader->isSmallVector)
302 {
303 newHeader = static_cast<SegmentHeader*>(Memory::allocate(sizeof(SegmentHeader) + newSize));
304 const auto minSize = min(newSize, static_cast<decltype(newSize)>(oldHeader->sizeBytes));
305 ::memcpy(newHeader, oldHeader, minSize + alignof(SegmentHeader));
306 newHeader->initDefaults();
307 newHeader->isFollowedBySmallVector = true;
308 }
309 else
310 {
311 newHeader = static_cast<SegmentHeader*>(Memory::reallocate(oldHeader, sizeof(SegmentHeader) + newSize));
312 }
313 if (newHeader)
314 {
315 newHeader->capacityBytes = static_cast<SegmentHeader::SizeType>(newSize);
316 }
317 return newHeader;
318}
319
320inline SC::SegmentHeader* SC::VectorAllocator::allocate(SegmentHeader* oldHeader, size_t numNewBytes, void* selfPointer)
321{
322 if (numNewBytes > SegmentHeader::MaxValue)
323 {
324 return nullptr;
325 }
326 if (oldHeader != nullptr)
327 {
328 if (oldHeader->isFollowedBySmallVector)
329 {
330 // If we were followed by a small vector, we check if that small vector has enough memory
331 SegmentHeader* followingHeader =
332 reinterpret_cast<SegmentHeader*>(static_cast<char*>(selfPointer) + alignof(SegmentHeader));
333 if (followingHeader->isSmallVector && followingHeader->capacityBytes >= numNewBytes)
334 {
335 return followingHeader;
336 }
337 }
338 else if (oldHeader->isSmallVector)
339 {
340 if (oldHeader->capacityBytes >= numNewBytes)
341 {
342 // shrink_to_fit on SmallVector pointing to internal buffer
343 return oldHeader;
344 }
345 }
346 }
347 SegmentHeader* newHeader = static_cast<SegmentHeader*>(Memory::allocate(sizeof(SegmentHeader) + numNewBytes));
348 if (newHeader)
349 {
350 newHeader->capacityBytes = static_cast<SegmentHeader::SizeType>(numNewBytes);
351 newHeader->initDefaults();
352 if (oldHeader != nullptr && oldHeader->isSmallVector)
353 {
354 newHeader->isFollowedBySmallVector = true;
355 }
356 }
357 return newHeader;
358}

360inline void SC::VectorAllocator::release(SegmentHeader* oldHeader)
361{
362 if (oldHeader->isSmallVector)
363 {
364 oldHeader->sizeBytes = 0;
365 }
366 else
367 {
368 Memory::release(oldHeader);
369 }

537template <typename T>
538template <typename U>
539bool SC::Vector<T>::append(Span<const U> src)
540{
541 const auto oldSize = size();
542 if (reserve(src.sizeInElements()))
543 {
544 for (auto& it : src)
545 {
546 if (not push_back(it))
547 {
548 break;
549 }
550 }
551 }
552 if (oldSize + src.sizeInElements() != size())
553 {
554 SC_ASSERT_RELEASE(resize(oldSize));
555 return false;
556 }
557 return true;
558}

611template <typename T>
612void SC::Vector<T>::moveAssign(Vector&& other)
613{
614 SegmentHeader* otherHeader = other.items != nullptr ? SegmentHeader::getSegmentHeader(other.items) : nullptr;
615 const bool otherIsSmallVector = otherHeader != nullptr && otherHeader->isSmallVector;
616 if (otherIsSmallVector)
617 {
618 // We can't "move" the small vector, so we just assign its items
619 clear();
620 (void)appendMove({other.items, other.size()});
621 other.clear();
622 }
623 else
624 {
625 const bool otherWasFollowedBySmallVector = otherHeader != nullptr && otherHeader->isFollowedBySmallVector;
626 if (otherHeader != nullptr)
627 {
628 // Before grabbing other.items we want to remember our state of "followed by/being a small vector"
629 const SegmentHeader* oldHeader = items != nullptr ? SegmentHeader::getSegmentHeader(items) : nullptr;
630 const bool shouldStillBeFollowedBySmallVector =
631 oldHeader != nullptr && (oldHeader->isFollowedBySmallVector || oldHeader->isSmallVector);
632 otherHeader->isFollowedBySmallVector = shouldStillBeFollowedBySmallVector;
633 }
634
635 destroy();
636 items = other.items;
637 if (otherWasFollowedBySmallVector)
638 {
639 // Other.items should become nullptr, but if it was followed by small vector, restore its link
640 // The Array<> holding the small buffer MUST be placed after the vector.
641 // We should advance by sizeof(Vector<T>) that is sizeof(void*) but on 32 bit we will still have
642 // some padding, as Array<> inherits from SegmentHeader, so it shares the same alignment (that is 8
643 // bytes on all platforms).
644 otherHeader = reinterpret_cast<SegmentHeader*>(reinterpret_cast<char*>(&other) + alignof(SegmentHeader));
645 other.items = otherHeader->getItems<T>();
646 }
647 else
648 {
649 other.items = nullptr;
650 }
651 }
