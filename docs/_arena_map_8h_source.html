24struct SC::ArenaMapKey
25{
26 private:
27 struct Generation
28 {
29 uint32_t used : 1;
30 uint32_t generation : 31;
31 Generation()
32 {
33 used = 0;
34 generation = 0;
35 }
36 bool operator==(const Generation other) const { return used == other.used and generation == other.generation; }
37 bool operator!=(const Generation other) const { return used != other.used or generation != other.generation; }
38 };
39 Generation generation;
40 uint32_t index;
41 friend struct ArenaMap<T>;
42 template <typename U>
43 friend struct ArenaMapKey;
44
45 public:
46 ArenaMapKey() { index = 0; }
47
48 bool isValid() const { return generation.used != 0; }
49
50 template <typename U>
51 ArenaMapKey<U> cast_to()
52 {
53 ArenaMapKey<U> key;
54 key.generation.used = generation.used;
55 key.generation.generation = generation.generation;
56 key.index = index;
57 return key;
58 }
59
60 template <typename U>
61 bool operator==(ArenaMapKey<U> other) const
62 {
63 return index == other.index and generation.used == other.generation.used and
64 generation.generation == other.generation.generation;
65 }
66
67 static constexpr uint32_t MaxGenerations = (uint32_t(1) << 31) - 1;
68 static constexpr uint32_t MaxIndex = 0xffffffff;

80struct SC::ArenaMap

82 using Key = ArenaMapKey<T>;

84 ArenaMap() {}

86 ~ArenaMap() { clear(); }

88 ArenaMap(const ArenaMap& other) { *this = other; }

90 ArenaMap(ArenaMap&& other) { *this = move(other); }

92 ArenaMap& operator=(const ArenaMap& other)
93 {
94 clear();
95 T* newItems = reinterpret_cast<T*>(Memory::allocate(other.itemsSize * sizeof(T)));
96
97 typename Key::Generation* newGenerations = reinterpret_cast<typename Key::Generation*>(
98 Memory::allocate(other.itemsSize * sizeof(typename Key::Generation)));
99 SC_ASSERT_RELEASE(newItems);
100 SC_ASSERT_RELEASE(newGenerations);
101 ::memset(newGenerations, 0, other.itemsSize * sizeof(typename Key::Generation));
102 items = newItems;
103 generations = newGenerations;
104 itemsSize = other.itemsSize;
105 for (size_t idx = 0; idx < other.itemsSize; ++idx)
106 {
107 if (other.generations[idx].used)
108 {
109 new (&items[idx], PlacementNew()) T(other.items[idx]);
110 }
111 generations[idx] = other.generations[idx];
112 }
113 numUsed = other.numUsed;
114 return *this;
115 }
116
117 ArenaMap& operator=(ArenaMap&& other)
118 {
119 clear();
120 items = other.items;
121 itemsSize = other.itemsSize;
122 other.items = nullptr;
123 other.itemsSize = 0;
124 generations = other.generations;
125 other.generations = nullptr;
126 numUsed = other.numUsed;
127 other.numUsed = 0;
128 return *this;
129 }

200 {

205 items[idx].~T();

226 [[nodiscard]] bool resize(size_t newSize)
227 {
228 if (numUsed != 0)
229 return false;
230 if (newSize > Key::MaxIndex)
231 return false;
232 if (items)
233 Memory::release(items);
234 items = nullptr;

236 Memory::release(generations);
237 generations = nullptr;

239 if (not newItems)
240 return false;

242 typename Key::Generation* newGenerations =
243 reinterpret_cast<typename Key::Generation*>(Memory::allocate(newSize * sizeof(typename Key::Generation)));
244 if (not newGenerations)
245 return false;
246 ::memset(newGenerations, 0, newSize * sizeof(typename Key::Generation));
247 generations = newGenerations;
248 itemsSize = newSize;
249 numUsed = 0;
250 return true;

291 template <typename ComparableToValue>
292 [[nodiscard]] bool containsValue(const ComparableToValue& value, Key* optionalKey = nullptr) const
293 {
294 for (size_t idx = 0; idx < itemsSize; ++idx)
295 {
296 if (generations[idx].used and items[idx] == value)
297 {
298 if (optionalKey)
299 {
300 optionalKey->index = static_cast<uint32_t>(idx);
301 optionalKey->generation = generations[idx];
302 }
303 return true;
304 }
305 }
306 return false;
307 }
308
309 [[nodiscard]] bool remove(Key key)
310 {
311 if (generations[key.index] != key.generation)
312 return false;
313 if (generations[key.index].generation + 1 <= Key::MaxGenerations)
314 {
315 generations[key.index].generation++;
316 }
317 else
318 {
319 // TODO: Should we give option of failing on generation overflow?
320 generations[key.index].generation++;
321 }
322 generations[key.index].used = 0;
323 items[key.index].~T();
324 numUsed--;
325 return true;
326 }
327
328 [[nodiscard]] T* get(Key key)
329 {
330 if (generations[key.index] != key.generation)
331 return nullptr;
332 return &items[key.index];
333 }
334
335 [[nodiscard]] const T* get(Key key) const
336 {
337 if (generations[key.index] != key.generation)
338 return nullptr;
339 return &items[key.index];
340 }

342 private:
343 T* items = nullptr;
344 size_t itemsSize = 0;
345
346 typename Key::Generation* generations = nullptr;
347

350 [[nodiscard]] Key allocateNewKeySlot()
351 {
352 for (size_t idx = 0; idx < itemsSize; ++idx)
353 {
354 if (generations[idx].used == 0)
355 {
356 generations[idx].used = 1;
357
358 Key key;
359 key.generation = generations[idx];
360 key.index = static_cast<uint32_t>(idx);
361 numUsed++;
362 return key;
363 }
364 }
365 return {};
366 }
