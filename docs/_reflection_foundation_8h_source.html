131template <typename T>
132static constexpr Sv ClNm()
133{
134 // clang-format off
135#if SC_COMPILER_CLANG || SC_COMPILER_GCC
136 const char* name = __PRETTY_FUNCTION__;
137 constexpr char separating_char = '=';
138 constexpr uint32_t skip_chars = 2;
139 constexpr uint32_t trim_chars = 1;
140 uint32_t length = 0;
141 const char* it = name;
142 while (*it != separating_char)
143 it++;
144 it += skip_chars;
145 while (it[length] != 0)
146 length++;
147 return Sv(it, length - trim_chars);
148#else
149 const char* name = __FUNCSIG__;
150 constexpr char separating_char = '<';
151 constexpr char ending_char = '>';
152 const char* it = name;
153 while (*it != separating_char)
154 it++;
155 auto itStart = it + 1;
156 while (*it != ending_char)
157 {
158 if (*it == ' ')
159 itStart = it + 1;
160 it++;
161 }
162 return Sv(itStart, static_cast<int>(it - itStart));
163#endif
164 // clang-format on
165}
