16template <typename SchemaBuilder>
17struct SchemaCompiler
18{
19 private:
20 using Type = typename SchemaBuilder::Type;
21 using TypeBuildFunction = typename Type::TypeBuildFunction;
22 using VirtualTablesType = decltype(SchemaBuilder::vtables);
23
25 template <uint32_t MAX_TOTAL_TYPES>
26 struct FlatFullResult
27 {
28 ArrayWithSize<Type, MAX_TOTAL_TYPES> types;
29 VirtualTablesType vtables;
30 };
31
33 template <uint32_t NUM_TYPES>
34 struct FlatTrimmedResult
35 {
36 ArrayWithSize<TypeInfo, NUM_TYPES> typeInfos;
37 ArrayWithSize<TypeStringView, NUM_TYPES> typeNames;
38 VirtualTablesType vtables;
39 };
40
41 template <uint32_t MAX_TYPES>
42 [[nodiscard]] static constexpr bool appendTypesTo(ArrayWithSize<Type, MAX_TYPES>& types, TypeBuildFunction build,
43 SchemaBuilder& builder)
44 {
45 // Let builder write to a slice of our available space in types array
46 const auto baseLinkID = types.size;
47 builder.currentLinkID = types.size;
48 builder.types = {types.values + types.size, MAX_TYPES - types.size};
49
50 if (build(builder))
51 {
52 // Set number of children for parent type and update types array size
53 const auto numberOfTypes = builder.currentLinkID - baseLinkID;
54 const auto numberOfChildren = numberOfTypes - 1;
55 if (numberOfChildren > static_cast<decltype(TypeInfo::numberOfChildren)>(MaxValue()))
56 return false;
57 if (not types.values[baseLinkID].typeInfo.setNumberOfChildren(numberOfChildren))
58 return false;
59
60 struct OrderByMemberOffset
61 {
62 constexpr bool operator()(const Type& a, const Type& b) const
63 {
64 return a.typeInfo.memberInfo.offsetInBytes < b.typeInfo.memberInfo.offsetInBytes;
65 }
66 };
67 if (types.values[baseLinkID].typeInfo.type == TypeCategory::TypeStruct and
68 types.values[baseLinkID].typeInfo.structInfo.isPacked)
69 {
70 // This is a little help for Binary Serialization, as packed structs end up serialized as is
71 Algorithms::bubbleSort(types.values + baseLinkID + 1, types.values + baseLinkID + 1 + numberOfChildren,
72 OrderByMemberOffset());
73 }
74 types.size += numberOfTypes;
75 return true;
76 }
77 return false;
78 }
79
80 template <uint32_t MAX_LINK_BUFFER_SIZE, uint32_t MAX_TOTAL_TYPES, typename Func>
81 constexpr static FlatFullResult<MAX_TOTAL_TYPES> compileAllTypesFor(Func func)
82 {
83 // Collect all types
84 FlatFullResult<MAX_TOTAL_TYPES> result;
85
86 SchemaBuilder container(result.types.values, MAX_TOTAL_TYPES);
87
88 ArrayWithSize<TypeBuildFunction, MAX_LINK_BUFFER_SIZE> alreadyVisitedTypes;
89 ArrayWithSize<uint32_t, MAX_LINK_BUFFER_SIZE> alreadyVisitedLinkID;
90 if (not appendTypesTo(result.types, func, container))
91 {
92 return {};
93 }
94
95 // Link all collected types
96 uint32_t typeIndex = 1;
97 while (typeIndex < result.types.size)
98 {
99 Type& type = result.types.values[typeIndex];
100 if (not type.typeInfo.isPrimitiveType() and type.typeInfo.needsLinking())
101 {
102 uint32_t outIndex = 0;
103 if (alreadyVisitedTypes.contains(type.typeBuild, &outIndex))
104 {
105 if (not type.typeInfo.setLinkIndex(alreadyVisitedLinkID.values[outIndex]))
106 return {};
107 }
108 else
109 {
110 if (not type.typeInfo.setLinkIndex(result.types.size))
111 return {};
112 if (not alreadyVisitedLinkID.push_back(result.types.size))
113 return {};
114 if (not alreadyVisitedTypes.push_back(type.typeBuild))
115 return {};
116 if (not appendTypesTo(result.types, type.typeBuild, container))
117 return {};
118 }
119 }
120 typeIndex++;
121 }
122 result.vtables = container.vtables;
123 return result;
124 }
125
126 public:
134 template <typename T, uint32_t MAX_LINK_BUFFER_SIZE = 20, uint32_t MAX_TOTAL_TYPES = 100>
135 static constexpr auto compile()
136 {
137 constexpr auto schema =
138 compileAllTypesFor<MAX_LINK_BUFFER_SIZE, MAX_TOTAL_TYPES>(&Reflect<T>::template build<SchemaBuilder>);
139 static_assert(schema.types.size > 0, "Something failed in compileAllTypesFor");
140
141 // Trim the returned FlatTrimmedResult only to the effective number of types
142 FlatTrimmedResult<schema.types.size> result;
143 for (uint32_t i = 0; i < schema.types.size; ++i)
144 {
145 result.typeInfos.values[i] = schema.types.values[i].typeInfo;
146 result.typeNames.values[i] = schema.types.values[i].typeName;
147 }
148 result.typeInfos.size = schema.types.size;
149 result.typeNames.size = schema.types.size;
150 result.vtables = schema.vtables;
151 return result;
152 }
153};

189 size_t offset)

198 TypeInfo::ArrayInfo arrayInfo)

200 return {TypeInfo(Reflect<T>::getCategory(), sizeof(T), numChildren, arrayInfo), name, &Reflect<T>::build};

205template <typename TypeVisitor>
206struct SchemaBuilder
207{
208 using Type = SchemaType<TypeVisitor>;
209
210 uint32_t currentLinkID;
211
212 WritableRange<Type> types;
213
214 constexpr SchemaBuilder(Type* output, const uint32_t capacity) : currentLinkID(0), types(output, capacity) {}
215
216 template <typename R, typename T, int N>
217 [[nodiscard]] constexpr bool operator()(uint8_t memberTag, R T::*field, const char (&name)[N], size_t offset)
218 {
219 currentLinkID++;
220 return types.writeAndAdvance(Type::createMember(memberTag, field, name, offset));
221 }
222
223 [[nodiscard]] constexpr bool addType(Type type)
224 {
225 currentLinkID++;
226 return types.writeAndAdvance(type);
227 }
228};
