441struct ConfigurePresets
442{
443 template <typename ConfigureFunction>
444 static Result generateAllPlatforms(ConfigureFunction configure, StringView projectName,
445 Build::Generator::Type generator, StringView targetDirectory,
446 StringView sourcesDirectory)
447 {
448 using namespace SC;
449 switch (generator)
450 {
451 case Build::Generator::VisualStudio2022: {
452 Build::Parameters parameters;
453 parameters.generator = Build::Generator::VisualStudio2022;
454 parameters.platforms = {Build::Platform::Windows};
455 parameters.architectures = {Build::Architecture::Arm64, Build::Architecture::Intel64};
456 Build::Definition definition;
457 SC_TRY(configure(definition, parameters, sourcesDirectory));
458 SC_TRY(definition.generate(projectName, parameters, targetDirectory));
459 }
460 break;
461 case Build::Generator::XCode: {
462 Build::Parameters parameters;
463 parameters.generator = Build::Generator::XCode;
464 parameters.platforms = {Build::Platform::MacOS};
465 parameters.architectures = {Build::Architecture::Arm64, Build::Architecture::Intel64};
466 Build::Definition definition;
467 SC_TRY(configure(definition, parameters, sourcesDirectory));
468 SC_TRY(definition.generate(projectName, parameters, targetDirectory));
469 }
470 break;
471
472 case Build::Generator::Makefile: {
473 Build::Parameters parameters;
474 parameters.generator = Build::Generator::Makefile;
475 parameters.platforms = {Build::Platform::MacOS, Build::Platform::Linux};
476 parameters.architectures = {Build::Architecture::Arm64, Build::Architecture::Intel64};
477 Build::Definition definition;
478 SC_TRY(configure(definition, parameters, sourcesDirectory));
479 SC_TRY(definition.generate(projectName, parameters, targetDirectory));
480 }
481 break;
482 }
483 return Result(true);
484 }
485};
