152template <typename... Types>
153inline bool SC::StringBuilder::append(StringView fmt, Types&&... args)
154{
155 if (not StringConverter::popNullTermIfExists(stringData, encoding))
156 {
157 return false;
158 }
159
160 StringFormatOutput sfo(encoding, stringData);
161 if (fmt.getEncoding() == StringEncoding::Ascii || fmt.getEncoding() == StringEncoding::Utf8)
162 {
163 // It's ok parsing format string '{' and '}' both for utf8 and ascii with StringIteratorASCII
164 // because on a valid UTF8 string, these chars are unambiguously recognizable
165 return StringFormat<StringIteratorASCII>::format(sfo, fmt, forward<Types>(args)...);
166 }
167 return false; // UTF16/32 format strings are not supported
168}
