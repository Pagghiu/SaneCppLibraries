242template <typename T, size_t N>
243struct Reflect<T[N]>
244{
245 static constexpr TypeCategory getCategory() { return TypeCategory::TypeArray; }
246
247 template <typename TypeVisitor>
248 [[nodiscard]] static constexpr bool build(TypeVisitor& builder)
249 {
250 using Type = typename TypeVisitor::Type;
251
252 // Add array type
253 constexpr bool isPacked = ExtendedTypeInfo<T>::IsPacked;
254 if (not builder.addType(Type::template createArray<T[N]>("Array", 1, TypeInfo::ArrayInfo{isPacked, N})))
255 return false;
256
257 // Add dependent item type
258 if (not builder.addType(Type::template createGeneric<T>()))
259 return false;
260
261 return true;
262 }
263};

276struct ReflectStruct
277{
278 using T = Type;
279
280 [[nodiscard]] static constexpr TypeCategory getCategory() { return TypeCategory::TypeStruct; }
281
282 template <typename TypeVisitor>
283 [[nodiscard]] static constexpr bool build(TypeVisitor& builder)
284 {
285 // Add struct type
286 if (not builder.addType(TypeVisitor::Type::template createStruct<T>()))
287 return false;
288
289 // Add all struct member
290 if (not Reflect<Type>::visit(builder))
291 return false;
292 return true;
293 }

297template <typename T>
298struct ExtendedStructTypeInfo
299{
300 size_t memberSizeSum = 0;
301 bool IsPacked = false;
302
303 constexpr ExtendedStructTypeInfo()
304 {
305 // Let's call operator() for each member
306 if (Reflect<T>::visit(*this))
307 {
308 // Structs are packed if all of its members are packed (checked in `operator()`)
309 // and in addition to that, summed size of all members is `==` `sizeof(T)`
310 IsPacked = memberSizeSum == sizeof(T);
311 }
312 }
313
314 template <typename R, int N>
315 constexpr bool operator()(int memberTag, R T::*member, const char (&name)[N], size_t offset)
316 {
317 SC_COMPILER_UNUSED(memberTag);
318 SC_COMPILER_UNUSED(name);
319 SC_COMPILER_UNUSED(member);
320 SC_COMPILER_UNUSED(offset);
321 if (not ExtendedTypeInfo<R>().IsPacked)
322 {
323 return false; // If a given type is not packed, let's stop iterating
324 }
325 memberSizeSum += sizeof(R);
326 return true; // continue iterating
327 }
328};
