340 template <class ForwardIt>
341 static constexpr ForwardIt MaxElement(ForwardIt first, ForwardIt last)
342 {
343 if (first == last)
344 return last;
345 ForwardIt largest = first;
346 for (++first; first != last; ++first)
347 {
348 if (*largest < *first)
349 largest = first;
350 }
351
352 return largest;
353 }
354 template <class T>
355 static constexpr T MaxElement(std::initializer_list<T> list)
356 {
357 return *MaxElement(list.begin(), list.end());
