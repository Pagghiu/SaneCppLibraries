119 enum class Type : uint8_t
120 {
121 LoopTimeout,
122 LoopWakeUp,
123 ProcessExit,
124 SocketAccept,
125 SocketConnect,
126 SocketSend,
127 SocketReceive,
128 SocketClose,
129 FileRead,
130 FileWrite,
131 FileClose,
132 FilePoll,
133 };
134
137 AsyncRequest(Type type) : state(State::Free), type(type), eventIndex(-1) {}
138
140
143 [[nodiscard]] Result stop();
144
145 protected:
146 [[nodiscard]] Result validateAsync();
147 [[nodiscard]] Result queueSubmission(AsyncEventLoop& eventLoop);
148
149 static void updateTime(AsyncEventLoop& loop);
150 AsyncEventLoop* eventLoop = nullptr;
151
152 private:
153 [[nodiscard]] static const char* TypeToString(Type type);
154 enum class State : uint8_t
155 {
156 Free, // not in any queue, this can be started with an async.start(...)
157 Setup, // when in submission queue waiting to be setup (after an async.start(...))
158 Submitting, // when in submission queue waiting to be activated (after a result.reactivateRequest(true))
159 Active, // when monitored by OS syscall or in activeWakeUps / activeTimeouts queues
160 Cancelling, // when in cancellation queue waiting for a cancelAsync (on active async)
161 Teardown // when in cancellation queue waiting for a teardownAsync (on non-active, already setup async)
162 };
163
164 friend struct AsyncEventLoop;
165
166 template <typename Lambda>
167 [[nodiscard]] static Result applyOnAsync(AsyncRequest& async, Lambda&& lambda);
168
169#if SC_CONFIGURATION_DEBUG
170 const char* debugName = "None";
171#endif
172 State state; // 1 byte
173 Type type; // 1 byte
174 int16_t flags = 0; // 2 bytes
175 int32_t eventIndex;
176
177 static constexpr int16_t Flag_RegularFile = 1 << 1;
178};
