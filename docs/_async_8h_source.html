179 [[nodiscard]] static const char* TypeToString(Type type);
180 enum class State : uint8_t
181 {
182 Free, // not in any queue, this can be started with an async.start(...)
183 Setup, // when in submission queue waiting to be setup (after an async.start(...))
184 Submitting, // when in submission queue waiting to be activated (after a result.reactivateRequest(true))
185 Active, // when monitored by OS syscall or in activeLoopWakeUps / activeTimeouts queues
186 Cancelling, // when in cancellation queue waiting for a cancelAsync (on active async)
187 Teardown // when in cancellation queue waiting for a teardownAsync (on non-active, already setup async)
188 };
189
190#if SC_CONFIGURATION_DEBUG
191 const char* debugName = "None";
192#endif
193 State state; // 1 byte
194 Type type; // 1 byte
195 int16_t flags; // 2 bytes
196 int32_t eventIndex; // 4 bytes
197};
