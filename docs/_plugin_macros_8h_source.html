60#define SC_PLUGIN_LINKER_DEFINITIONS \

62 extern "C" void bzero(void* s, SC::size_t n) \
63 { \
64 unsigned char* p = reinterpret_cast<unsigned char*>(s); \
65 while (n-- > 0) \
66 { \
67 *p++ = 0; \
68 } \
69 } \
70 extern "C" int memcmp(const void* s1, const void* s2, SC::size_t n) \
71 { \
72 const unsigned char* p1 = reinterpret_cast<const unsigned char*>(s1); \
73 const unsigned char* p2 = reinterpret_cast<const unsigned char*>(s2); \
74 while (n--) \
75 { \
76 if (*p1 < *p2) \
77 { \
78 return -1; \
79 } \
80 else if (*p1 > *p2) \
81 { \
82 return 1; \
83 } \
84 } \
85 return 0; \
86 }
