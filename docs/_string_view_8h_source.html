702template <typename StringIterator1, typename StringIterator2>
703constexpr bool SC::StringView::equalsIterator(StringIterator1 t1, StringIterator2 t2, size_t& points)
704{
705 StringCodePoint c1 = 0;
706 StringCodePoint c2 = 0;
707 while (t1.advanceRead(c1) and t2.advanceRead(c2))
708 {
709 if (c1 != c2)
710 {
711 return false;
712 }
713 points++;
714 }
715 return t1.isAtEnd() and t2.isAtEnd();
716}

718template <typename StringIterator>
719constexpr bool SC::StringView::equalsIterator(StringView other, size_t& points) const
720{
721 auto it = getIterator<StringIterator>();
722 switch (other.getEncoding())
723 {
724 case StringEncoding::Ascii: return equalsIterator(it, other.getIterator<StringIteratorASCII>(), points);
725 case StringEncoding::Utf8: return equalsIterator(it, other.getIterator<StringIteratorUTF8>(), points);
726 case StringEncoding::Utf16: return equalsIterator(it, other.getIterator<StringIteratorUTF16>(), points);
727 }
728 Assert::unreachable();
729}

742[[nodiscard]] constexpr bool SC::StringView::operator==(StringView other) const
743{
744 if (hasCompatibleEncoding(other))
745 {
746 if (textSizeInBytes != other.textSizeInBytes)
747 return false;
748 if (__builtin_is_constant_evaluated())
749 {
750 auto it1 = text;
751 auto it2 = other.text;
752 auto sz = textSizeInBytes;
753 for (size_t idx = 0; idx < sz; ++idx)
754 if (it1[idx] != it2[idx])
755 return false;
756 }
757 else
758 {
759 if (text == nullptr)
760 {
761 return other.textSizeInBytes == 0;
762 }
763 if (other.text == nullptr)
764 {
765 return textSizeInBytes == 0;
766 }
767 return memcmp(text, other.text, textSizeInBytes) == 0;
768 }
769 }
770 size_t commonOverlappingPoints = 0;
771 return fullyOverlaps(other, commonOverlappingPoints);
772}

774constexpr bool SC::StringView::fullyOverlaps(StringView other, size_t& commonOverlappingPoints) const
775{
776 commonOverlappingPoints = 0;
777 switch (getEncoding())
778 {
779 case StringEncoding::Ascii: return equalsIterator<StringIteratorASCII>(other, commonOverlappingPoints);
780 case StringEncoding::Utf8: return equalsIterator<StringIteratorUTF8>(other, commonOverlappingPoints);
781 case StringEncoding::Utf16: return equalsIterator<StringIteratorUTF16>(other, commonOverlappingPoints);
782 }
783 Assert::unreachable();
784}

816template <typename StringIterator>
817inline SC::StringView SC::StringView::fromIterators(StringIterator from, StringIterator to)
818{
819 const ssize_t numBytes = to.bytesDistanceFrom(from);
820 if (numBytes >= 0)
821 {
822 StringIterator fromEnd = from;
823 fromEnd.setToEnd();
824 if (fromEnd.bytesDistanceFrom(to) >= 0) // If current iterator of to is inside from range
825 return StringView({from.getCurrentIt(), static_cast<size_t>(numBytes)}, false,
826 StringIterator::getEncoding());
827 }
828 return StringView(); // TODO: Make StringView::fromIterators return bool to make it fallible
829}

849constexpr SC::StringView SC::StringView::sliceStartBytes(size_t start) const
850{
851 if (start < sizeInBytes())
852 return sliceStartLengthBytes(start, sizeInBytes() - start);
853 SC_ASSERT_RELEASE(start < sizeInBytes());
854 return StringView({text, 0}, false, getEncoding());
855}

857constexpr SC::StringView SC::StringView::sliceStartEndBytes(size_t start, size_t end) const
858{
859 if (end >= start)
860 return sliceStartLengthBytes(start, end - start);
861 SC_ASSERT_RELEASE(end >= start);
862 return StringView({text, 0}, false, getEncoding());
863}
864
865constexpr SC::StringView SC::StringView::sliceStartLengthBytes(size_t start, size_t length) const
866{
867 if (start + length > sizeInBytes())
868 {
869 SC_ASSERT_RELEASE(start + length > sizeInBytes());
870 return StringView({text, 0}, false, getEncoding());
871 }
872 return StringView({text + start, length}, hasNullTerm and (start + length == sizeInBytes()), getEncoding());
873}
