683template <typename StringIterator1, typename StringIterator2>
684constexpr bool SC::StringView::equalsIterator(StringIterator1 t1, StringIterator2 t2, size_t& points)
685{
686 StringCodePoint c1 = 0;
687 StringCodePoint c2 = 0;
688 while (t1.advanceRead(c1) and t2.advanceRead(c2))
689 {
690 if (c1 != c2)
691 {
692 return false;
693 }
694 points++;
695 }
696 return t1.isAtEnd() and t2.isAtEnd();
697}

699template <typename StringIterator>
700constexpr bool SC::StringView::equalsIterator(StringView other, size_t& points) const
701{
702 auto it = getIterator<StringIterator>();
703 switch (other.getEncoding())
704 {
705 case StringEncoding::Ascii: return equalsIterator(it, other.getIterator<StringIteratorASCII>(), points);
706 case StringEncoding::Utf8: return equalsIterator(it, other.getIterator<StringIteratorUTF8>(), points);
707 case StringEncoding::Utf16: return equalsIterator(it, other.getIterator<StringIteratorUTF16>(), points);
708 }
709 Assert::unreachable();
710}

723[[nodiscard]] constexpr bool SC::StringView::operator==(StringView other) const
724{
725 if (hasCompatibleEncoding(other))
726 {
727 if (textSizeInBytes != other.textSizeInBytes)
728 return false;
729 if (__builtin_is_constant_evaluated())
730 {
731 auto it1 = text;
732 auto it2 = other.text;
733 auto sz = textSizeInBytes;
734 for (size_t idx = 0; idx < sz; ++idx)
735 if (it1[idx] != it2[idx])
736 return false;
737 }
738 else
739 {
740 if (text == nullptr)
741 {
742 return other.textSizeInBytes == 0;
743 }
744 if (other.text == nullptr)
745 {
746 return textSizeInBytes == 0;
747 }
748 return memcmp(text, other.text, textSizeInBytes) == 0;
749 }
750 }
751 size_t commonOverlappingPoints = 0;
752 return fullyOverlaps(other, commonOverlappingPoints);
753}

755constexpr bool SC::StringView::fullyOverlaps(StringView other, size_t& commonOverlappingPoints) const
756{
757 commonOverlappingPoints = 0;
758 switch (getEncoding())
759 {
760 case StringEncoding::Ascii: return equalsIterator<StringIteratorASCII>(other, commonOverlappingPoints);
761 case StringEncoding::Utf8: return equalsIterator<StringIteratorUTF8>(other, commonOverlappingPoints);
762 case StringEncoding::Utf16: return equalsIterator<StringIteratorUTF16>(other, commonOverlappingPoints);
763 }
764 Assert::unreachable();
765}

797template <typename StringIterator>
798inline SC::StringView SC::StringView::fromIterators(StringIterator from, StringIterator to)
799{
800 const ssize_t numBytes = to.bytesDistanceFrom(from);
801 if (numBytes >= 0)
802 {
803 StringIterator fromEnd = from;
804 fromEnd.setToEnd();
805 if (fromEnd.bytesDistanceFrom(to) >= 0) // If current iterator of to is inside from range
806 return StringView({from.getCurrentIt(), static_cast<size_t>(numBytes)}, false,
807 StringIterator::getEncoding());
808 }
809 return StringView(); // TODO: Make StringView::fromIterators return bool to make it fallible
810}

830constexpr SC::StringView SC::StringView::sliceStartBytes(size_t start) const
831{
832 if (start < sizeInBytes())
833 return sliceStartLengthBytes(start, sizeInBytes() - start);
834 SC_ASSERT_RELEASE(start < sizeInBytes());
835 return StringView({text, 0}, false, getEncoding());
836}
837
838constexpr SC::StringView SC::StringView::sliceStartEndBytes(size_t start, size_t end) const
839{
840 if (end >= start)
841 return sliceStartLengthBytes(start, end - start);
842 SC_ASSERT_RELEASE(end >= start);
843 return StringView({text, 0}, false, getEncoding());
844}
845
846constexpr SC::StringView SC::StringView::sliceStartLengthBytes(size_t start, size_t length) const
847{
848 if (start + length > sizeInBytes())
849 {
850 SC_ASSERT_RELEASE(start + length > sizeInBytes());
851 return StringView({text, 0}, false, getEncoding());
852 }
853 return StringView({text + start, length}, hasNullTerm and (start + length == sizeInBytes()), getEncoding());
854}
