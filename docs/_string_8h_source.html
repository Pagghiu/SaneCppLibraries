161inline bool SC::String::assign(StringView sv)
162{
163 encoding = sv.getEncoding();
164 const size_t length = sv.sizeInBytes();
165 const size_t numZero = StringEncodingGetSize(encoding);
166 if (not data.resizeWithoutInitializing(length + numZero))
167 return false;
168 if (sv.isNullTerminated())
169 {
170 memcpy(data.items, sv.bytesWithoutTerminator(), length + numZero);
171 }
172 else
173 {
174 memcpy(data.items, sv.bytesWithoutTerminator(), length);
175 for (size_t idx = 0; idx < numZero; ++idx)
176 {
177 data.items[length + idx] = 0;
178 }
179 }
180 return true;
181}

183inline SC::StringView SC::String::view() const
184{
185 const bool isEmpty = data.isEmpty();
186 const char* items = isEmpty ? nullptr : data.items;
187 return StringView({items, isEmpty ? 0 : data.size() - StringEncodingGetSize(encoding)}, not isEmpty, encoding);
188}
